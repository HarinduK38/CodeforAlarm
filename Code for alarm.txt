int piezoPin = 8;

void setup() {
  Serial.begin(9600);
  pinMode(3, INPUT);
}

void loop() {
   if (digitalRead(3) == HIGH) {
    Serial.println("Active");
    tone(piezoPin, 1000, 500);
    tone(piezoPin, 1000, 500);
    delay(1000);
  }
  if (digitalRead(3) == LOW) {
    Serial.println("Active");
    tone(piezoPin, 1000, 500);
    delay(1000);
  }
  else {
    Serial.println("Inactive");
  }
  delay(1000);
}
